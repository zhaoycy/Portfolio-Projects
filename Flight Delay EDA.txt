convert_to_posix <- function(time) {
  hour <- time %/% 100
  minute <- time %% 100
  as.POSIXct(sprintf("%02d:%02d", hour, minute), format="%H:%M")
}
data_clean$schedtime_posix <- sapply(data_clean$schedtime, convert_to_posix)
